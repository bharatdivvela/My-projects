Pool Area¶

In [20]:
#H0: the SalePrice means of two groups are indifferent
#Ha: the SalePrice means of two groups are different
observed_mean_diff = traindf[traindf['PoolArea'] > 0]['SalePrice'].mean() - traindf[traindf['PoolArea'] == 0]['SalePrice'].mean()
print('Observed diff:',observed_mean_diff)
mean_diff = []
for i in range(1000):
    mean_pool = traindf[traindf['PoolArea'] > 0]['SalePrice'].sample(frac=1,replace=True).mean()
    mean_no_pool = traindf[traindf['PoolArea'] == 0]['SalePrice'].sample(frac=1,replace=True).mean()
    mean_diff.append(mean_pool - mean_no_pool)

confidence_interval = np.percentile(mean_diff,[0.025,0.975])
print(confidence_interval)

def modelfit(alg, train, test, predictors, target, IDcol, filename):
    #Fit the algorithm on the data
    alg.fit(train[predictors], train[target])
        
    #Predict training set:
    train_predictions = alg.predict(train[predictors])
    #Perform cross-validation:
    cv_score = cross_val_score(alg, train[predictors],(train[target]) , cv=10, scoring='neg_mean_squared_error')
    cv_score = np.sqrt(np.abs(cv_score))
    
    #Print model report:
    print('RMSE:', np.sqrt(metrics.mean_squared_error((np.exp(train[target])).values, train_predictions)))
    print('CV Score : Mean:', np.mean(cv_score) , '| Min:', np.min(cv_score), '| Max:',np.max(cv_score))
    plt.scatter(np.exp(train[target]).values, np.exp(train_predictions))
    plt.xlabel('Actual SalePrice')
    plt.ylabel('Predicted SalePrice')
    #Predict on testing data:
    test[target] = np.exp(alg.predict(test[predictors]))
    
    #Export submission file:
    IDcol.append(target)
    submission = pd.DataFrame({ x: test[x] for x in IDcol})
    submission.to_csv(filename, index=False)
